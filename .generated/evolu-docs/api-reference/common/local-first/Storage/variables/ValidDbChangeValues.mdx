const ValidDbChangeValues: brand(
  "ValidDbChangeValues",
  DbChangeValues,
  (value) => {
    const invalidColumns = systemColumnsWithId.filter((key) => key in value);
    if (invalidColumns.length > 0)
      return err<ValidDbChangeValuesError>({
        type: "ValidDbChangeValues",
        value,
        invalidColumns,
      });

    return ok(value);
  },
);
