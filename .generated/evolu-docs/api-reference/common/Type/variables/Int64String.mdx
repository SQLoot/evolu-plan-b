const Int64String: brand(
  "Int64",
  NonEmptyTrimmedString,
  (value) =>
    trySync(
      () => {
        const maybeInt = globalThis.BigInt(value);
        Int64.orThrow(maybeInt);
        return value;
      },
      (): Int64StringError => ({ type: "Int64String", value }),
    ),
);
